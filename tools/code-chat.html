Quick Start

# Clone and install

cd code-chat
npm install

# Configure your project in config.json
{
  "port": 3333,
  "auth": { "enabled": true, "keys": { ... } },
  "projects": [{
    "id": "myapp",
    "name": "My App",
    "path": "/path/to/project",
    "devCommand": "npm run dev",
    "previewPort": 3001
  }]
}

npm start
